#UNIVARIATE MODELS
#sst
model0 <- scam (occurrenceStatus ~  s(BO_sstmean, k=8,bs="cv"), family=binomial(link="logit"), data=data)
summary(model0)
plotmo(model0,level = 0.95, pt.col=8)

#chlorophyll
model0 <- scam (occurrenceStatus ~  s(BO2_chlomean_ss, k=8,bs="cv"), family=binomial(link="logit"), data=data)
summary(model0)
plotmo(model0,level = 0.95, pt.col=8)

model0 <- scam (occurrenceStatus ~  s(BO2_salinitymean_ss, k=8,bs="cv"), family=binomial(link="logit"), data=data,sp=0.00001)
summary(model0)
plotmo(model0,level = 0.95, pt.col=8)


# model with the 3 variables
model <- scam (occurrenceStatus ~  s(BO_sstmean, k=8,bs="cv")+ s(BO2_salinitymean_ss, k=8,bs="cv")+s(BO2_chlomean_ss, k=8,bs="cv"), family=binomial(link="logit"), data=data,sp=rep(0.00001,3))


#Check model which contains all variables
summary(model)
scam.check(model)
plot(model)
plotmo(model,level = 0.95, pt.col=8)

# Save the model
selected_model<-model
save(list=c("selected_model"), file = ("models/model.RData"))

Code to build SC-GAM model default sp= NULL, k = 8 and plevel = 0.05

  # apply the model selection function (no limits in the number of terms)
  model_SCGAM <- try(modsel.scam(basef="occurrenceStatus ~ 1", vars=vars, dat=data,sp=0.00001), silent=T) # by default aic.tol=2
#NOTA INTERNA: ¿queremos añadir el if else y asignar un valor de sp en caso de que sp NULL de error?

check results

model_SCGAM$svars
sapply(model_SCGAM$smod, AIC)
sapply(model_SCGAM$smod, function(x) summary(x)$dev.expl)
lapply(model_SCGAM$smod, formula)
lapply(model_SCGAM$smod, summary)

#Select the last model of the list which is the best model according to AIC and plevel criteria
selected_model <- model_SCGAM$smod[[length(model_SCGAM$smod)]]
selected_model$sp

# see specifically the summary and the plots of the final model after the selection
# note that it is the last element of the list of models

scam.check(selected_model)
plot(selected_model)
